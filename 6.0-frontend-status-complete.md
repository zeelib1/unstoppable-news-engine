1. Moved to environment variables

**Recommended:**
```javascript
// Create composable: composables/useApi.ts
export const useApi = () => {
  const config = useRuntimeConfig()
  const baseURL = config.public.apiUrl || 'http://localhost:3001'
  
  return {
    baseURL,
    fetch: (path: string, options?: any) => {
      return fetch(`${baseURL}${path}`, options)
    }
  }
}
```

